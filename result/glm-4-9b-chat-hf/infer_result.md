```python
def two_sum(nums, target):
    num_to_index = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in num_to_index:
            return [num_to_index[complement]], i]
        num_to_index[num] = i
    return []

# 示例
nums = [2, 7, 11, 15]
target = 9

print(two_sum(nums, target)))
